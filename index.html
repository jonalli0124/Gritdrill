function preload() { // Load the drill bit images this.load.image('drill1', './Drill1.png'); this.load.image('drill2', './Drill2.png'); this.load.image('drill3', './Drill3.png'); this.load.image('drill4', './Drill4.png'); this.load.image('drill5', './Drill5.png'); } function create() { // Set background color this.cameras.main.setBackgroundColor('#e0e0e0'); // Add the drill bit sprite drillBit = this.physics.add.sprite(400, 300, 'drill1'); // Create animation for the drill bit using frames this.anims.create({ key: 'drilling', frames: [ { key: 'drill1' }, { key: 'drill2' }, { key: 'drill3' }, { key: 'drill4' }, { key: 'drill5' } ], frameRate: 10, repeat: -1 }); // Play the animation drillBit.anims.play('drilling'); } const config = { type: Phaser.AUTO, width: 800, height: 600, scene: { preload: preload, create: create } }; const game = new Phaser.Game(config);
